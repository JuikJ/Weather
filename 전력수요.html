def calculate_average_power_usage(data, temperature_range):
    filtered_data = data[(data['electric_train.nph_ta'] >= temperature_range[0]) & 
                        (data['electric_train.nph_ta'] <= temperature_range[1])]
    average_power_usage = filtered_data['electric_train.sum_load'].mean()
    return average_power_usage

temperature_ranges = [(-float('inf'), -20), (-20, -10), (-9, 0), (1, 10), (11, 20), (21, 30), (30, float('inf'))]

#      
average_usages = []
for temp_range in temperature_ranges:
    average_usage = calculate_average_power_usage(train, temp_range)
    average_usages.append(average_usage)
#
plt.figure(figsize=(10, 3))
temperature_range_labels = [f"{temp_range[0] }C ~{temp_range[1]}C " for temp_range in temperature_ranges]
plt.plot(temperature_range_labels, average_usages)
plt.grid(axis='y')  
plt.title('sum load by ta')
plt.xlabel('temperature')
plt.ylabel('sumload')
plt.ylim(3000,15000)
plt.xticks(rotation=45) 
plt.show()
